int serialNumber = 1;
class Player
{
int playerInstanceID;
public: 
//constructor
Player()
{
playerInstanceID = serialNumber++;
cout << "Player: " << playerInstanceID << " created..\n";
    
}
//destructor
~Player()
{
cout << "Player: " << playerInstanceID << " destroyed..\n";
}
//for debug and demo purposes.. to print the player instance number
string ToString()
{
return "Player " + std::to_string(playerInstanceID) + "\n";
}
};
