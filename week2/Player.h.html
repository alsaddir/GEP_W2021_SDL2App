#pragma once
#include <string>
using namespace std;
class Player
{
private:
string playerName;
float health;
int score;
public:
Player(string pName) { playerName = pName; health = 100; }
~Player() {  }
/****Getters/accessors *****/
string getPlayerName() { return playerName; }
float  getHealth() { return health; }
int    getScore() { return score; }
//Member functions
void AddToScore(int points) { score += points; }
void ReduceHealth(float h) { health -= h; }
void Reset() { health = 100; score = 0; }
};
