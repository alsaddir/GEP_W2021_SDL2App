#pragma once
#include <iostream>
#include <string>
using namespace std;
enum DOOR_TYPE{ TREASURE, ENEMY};
class Door
{
private:
DOOR_TYPE doorType; 
int value; //this contains either the treasure value ($$$), or damage, depending on the door type
public:
Door() {}
~Door() {}
void InitializeDoor(DOOR_TYPE dt, int val)
{
doorType = dt;
value = val;
}
void Print() {
cout << "Door Type: " << doorType << " | value: " 
<< value << endl; 
}
};
