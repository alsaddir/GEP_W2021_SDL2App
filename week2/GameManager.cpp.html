#include "GameManager.h"
#include "Random.h"
 GameManager::GameManager(int totalDoorsToGenerate)
{
totalDoors = totalDoorsToGenerate;
openedDoors = 0;
//allocate memory
doors = new Door[totalDoors];
for (int i = 0; i < totalDoorsToGenerate; i++)
{
DOOR_TYPE dt = (DOOR_TYPE) GEP::RandomRange(0, 1);
int randVal = 0;
if (dt == 0)//treasure type
randVal = GEP::RandomRange(100, 500);
else //generate enemy damage
randVal = GEP::RandomRange(10, 30);
doors[i].InitializeDoor(dt, randVal);
}
}
 GameManager::~GameManager()
 {
delete[] doors;
doors = nullptr;
 }
 void GameManager::PrintAllDoors()
 {
for (int i = 0; i < totalDoors; i++)
doors[i].Print();
 }
