#include "stdafx.h"
#include <iostream>
#include <iomanip>
using namespace std;
//Raw pointer example:
/**
This program totals and averages the number of players playing your multiplayer game over 
any number of days.
The figures are stored in a dynamically allocated array.
**/
int main()
{
double *players = nullptr;    //to dynamically allocate an array
double total = 0.0;     
double average;  
int numDays; 
cout << "How many days of player data do you wish to process? ";
cin >> numDays;
//Dynamically allocate an array large enough to hold that many days of playertotals amount
players = new double[numDays]; //allocates memory
cout << "Enter the player figures below. (How many players played your game each day): ";
//Get user input through for loop
for (int i = 0; i < numDays; i++)
{
cout << "Day " << (i + 1) << ": ";
cin >> players[i];
total += players[i];
}
average = total / numDays;
//Display the results:
cout << setprecision(2) << fixed;
cout << "Total Players: " << total << endl;
cout << "Average Players/day: " << average << endl;
//Free the dynamically allocated memory
delete[] players;
players = nullptr;
    return 0;
}
